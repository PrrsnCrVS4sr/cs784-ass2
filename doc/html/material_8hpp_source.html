33 **/
34 class material_t
35 {
36 private:
37 /// Name of the material.
38 std::string name;
39 public:
40 /// Constructor.
41 material_t(std::string _name);
42
43 /// Destructor.
44 virtual ~material_t();
45
46 /// Returns the name of the material.
47 std::string get_name(void) const;
48
49 /// Returns the diffuse coefficient (kd) for direct illumination.
50 virtual color_t get_diffuse(void) const = 0;
51 /// Returns the specular reflection coefficient (ks) for direct illumination.
52 virtual color_t get_specular(void) const = 0;
53 /// Returns the transmission coefficient (kt). The reflection coefficient (kr) can be 1-kt.
54 virtual color_t get_transmit(void) const = 0;
55 /// Returns the reflection coefficient (kr), which can be 1-kt.
56 virtual color_t get_reflect(void) const = 0;
57
58 /// Returns the coefficient of refraction
59 virtual double get_eta(void) const = 0;
60 /// Returns the shininess value
61 virtual double get_shininess(void) const = 0;
62
63 /// Returns true if surface reflects, else returns false.
64 virtual bool get_is_reflect(void) const = 0;
65 /// Returns false if surface transmits, else returns false.
66 virtual bool get_is_transmit(void) const = 0;
67
68 /// Prints information about the material to the stream.
69 virtual void print(std::ostream &stream) const = 0;
70 };
