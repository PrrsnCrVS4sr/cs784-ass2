29/**
30 * The image2cam transformation is similar to the modeling-viewing pipeline transformation that opengl
31 * does to get from the VCS to the CCS.
32 *
33 * The cam2world transformation is set to identity by default. This is the WCS to VCS transform.
34 *
35 * Must ensure that this is called at least once with valid parameters, before the camera can
36 * act as a producer of primary rays.
37*/
38void camera_t::init(void)
39{
40 double recip = 1.0 / (far - near);
41 double cot = 1.0 / std::tan(fov / 2.0);
42
43 Eigen::Matrix4d perspective;
44 perspective <<
45 cot, 0, 0, 0,
46 0, cot, 0, 0,
47 0, 0, far * recip, -near * far * recip,
48 0, 0, 1, 0;
49
50
51
52 image2cam = transform_t(
53 Eigen::DiagonalMatrix<double, 3>(Vector3d(-0.5, -0.5 * aspect, 1.0)) *
54 Eigen::Translation<double, 3>(-1.0, -1.0/aspect, 0.0) * perspective).inverse();
55
56 cam2world = transform_t();
57}

88color_t camera_t::sample_ray(ray_t &ray, const Vector2f& _pixelpos) const
89{
90 Vector3d pt(_pixelpos.x(), _pixelpos.y(), 0.0);
91 Vector3d near_p= image2cam.transform_point(pt);
92 Vector3d d = near_p.normalized();
93 double onebyz = 1.0/d.z();
94
95 ray.origin = cam2world.transform_point(Vector3d(0.0,0.0,0.0));
96 ray.direction = cam2world * d;
97
98 ray.mint = near * onebyz;
99 ray.maxt = far * onebyz;
100
101 return color_t(1.0);
102}
