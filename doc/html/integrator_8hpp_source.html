40 **/
41 class integrator_t

43 public:

45 /// This may or may not be used by the implementing class
46 int depth;
47
48 /// Constructor
49 integrator_t():depth(1) { }
50 /// Constructor
51 integrator_t(int _d):depth(_d) { }

53 virtual ~integrator_t() { }
54
55 ///Returns a name for the integrator.
56 virtual std::string get_name(void) const = 0;
57
58 ///Returns the radiance estimate along ray _ray, for scene scn, at bounce depth d.
