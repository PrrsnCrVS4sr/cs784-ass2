50color_t image_t::get_pixel(unsigned int _x, unsigned int _y) const
51{
52 int pos=(_y)*width*3+(_x)*3;
53 double r=double(data[pos])/255.0;
54 double g=double(data[pos+1])/255.0;
55 double b=double(data[pos+2])/255.0;
56 return color_t(r,g,b);
57}

59void image_t::set_pixel(unsigned int _x, unsigned int _y, color_t _col)
60{
61 int pos=(_y)*width*3+(_x)*3;
62 char r = to_char(_col.r());
63 char g = to_char(_col.g());
64 char b = to_char(_col.b());
65 data[pos]=r; data[pos+1]=g; data[pos+2]=b;
66}
