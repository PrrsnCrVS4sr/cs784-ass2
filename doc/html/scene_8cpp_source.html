167bool scene_t::parse_scenefile(void)
168{
169 XMLError load_okay = scenexml.LoadFile(scenefile.c_str());
170
171 if (load_okay!= tinyxml2::XML_SUCCESS)
172 {
173 throw std::runtime_error("Cannot read scene file.");
174 }
175
176 XMLElement* doc_root = scenexml.RootElement();
177 if (!doc_root || !XMLUtil::StringEqual(doc_root->Name(), "rt-scene"))
178 throw std::runtime_error("Scene file is not a valid scene description");
179
180 cam = parse_camera(doc_root);
181 cam->print(std::cerr);
182
183 img = parse_image(doc_root);
184 img->print(std::cerr);
185
186 cam->set_aspect(img->get_aspect());
187
188 intg = parse_integrator(doc_root);
189
190 nummats = parse_materials(doc_root);
191 std::list<material_t*>::iterator it;
192 for (it=mats.begin(); it!=mats.end(); it++)
193 {
194 (*it)->print(std::cerr);
195 }
196
197 numobjs = parse_objects(doc_root, mats);
198 std::vector<object_t*>::iterator oit;
199 for (oit=objs.begin(); oit!=objs.end(); oit++)
200 {
201 (*oit)->print(std::cerr);
202 }
203
204 numlits = parse_lights(doc_root);
205 std::list<light_t*>::iterator lit;
206 for (lit=lits.begin(); lit!=lits.end(); lit++)
207 {
208 (*lit)->print(std::cerr);
209 }
210
211 return true;
212
213}

231int scene_t::parse_objects(XMLElement* _elm, const std::list<material_t*>& matlist)
232{
233 objs.clear();
234
235 XMLElement* elm_obj = _elm->FirstChildElement("objects");
236 if (!elm_obj)
237 throw std::runtime_error("No objects found in scene file.");
238
239 int numobj=1;
240 XMLElement* elm_child = elm_obj->FirstChildElement();
241 while (elm_child)
242 {
243 std::string name(elm_child->Name());
244
245 if (name == "sphere")
246 objs.push_back(parse_object_sphere(elm_child, matlist));
247 else
248 throw std::invalid_argument("Invalid object in scene file.");
249
250 elm_child = elm_child->NextSiblingElement();
251 numobj++;
252 }
253
254 if (objs.empty())
255 throw std::invalid_argument("Could not parse object.");
256 else
257 return numobj;
258}
